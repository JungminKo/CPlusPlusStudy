다음 main함수에서 필요로 하는 swap 함수를 오버로딩 해서 구현해보자.

```C++
#include <iostream>

void swap(int* pt1, int* pt2)
{
	int temp = *pt1;
	*pt1 = *pt2;
	*pt2 = temp;
}

void swap(char* pt1, char* pt2)
{
	char temp = *pt1;
	*pt1 = *pt2;
	*pt2 = temp;
}

void swap(double* pt1, double* pt2)
{
	double temp = *pt1;
	*pt1 = *pt2;
	*pt2 = temp;
}

int main(void)
{
	int num1 = 20, num2 = 30;
	swap(&num1, &num2);
	std::cout << num1 << ' ' << num2 << std::endl;

	char ch1 = 'A', ch2 = 'Z';
	swap(&ch1, &ch2);
	std::cout << ch1 << ' ' << ch2 << std::endl;

	double dbl1 = 1.111, dbl2 = 5.555;
	swap(&dbl1, &dbl2);
	std::cout << dbl1 << ' ' << dbl2 << std::endl;
		
	return 0;
}
```
